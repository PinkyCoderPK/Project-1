#include<bits/stdc++.h>
using namespace std;

string s;
queue<int> a;
int x;

int main()
{
    // Week 3 - Simulation Queue
    //freopen("code.inp","r",stdin);
    //freopen("code.out","w",stdout);
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);

    while ( cin >> s )
    {
        if ( s == "#" ) break;
        if ( s == "POP" )
        {
            if( a.empty() ) cout << "NULL\n";
            else
            {
                cout << a.front() <<"\n";
                a.pop();
            }
        }
        else
        {
            cin >> x;
            a.push(x);
        }
    }
}
